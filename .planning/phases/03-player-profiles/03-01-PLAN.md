create table public.player_profiles (
  id uuid primary key default gen_random_uuid(),
  community_id uuid references public.communities(id) on delete cascade not null,
  user_id uuid references auth.users(id) on delete cascade not null,
  display_name text,
  phone text,
  bio text,
  avatar_url text,
  self_skill_level text check (self_skill_level in ('beginner', 'intermediate', 'advanced')),
  utr numeric(5,2),
  coaching_bio text,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique (community_id, user_id)
);

alter table public.player_profiles enable row level security;

create policy "player_profiles_select"
on public.player_profiles for select to authenticated
using (community_id = ((select auth.jwt()) ->> 'community_id')::uuid);

create policy "player_profiles_insert"
on public.player_profiles for insert to authenticated
with check (
  community_id = ((select auth.jwt()) ->> 'community_id')::uuid
  and user_id = (select auth.uid())
);

create policy "player_profiles_update"
on public.player_profiles for update to authenticated
using (
  community_id = ((select auth.jwt()) ->> 'community_id')::uuid
  and user_id = (select auth.uid())
);

-- 3. coach_assessments table
create table public.coach_assessments (
  id uuid primary key default gen_random_uuid(),
  community_id uuid references public.communities(id) on delete cascade not null,
  subject_member_id uuid references public.community_members(id) on delete cascade not null,
  coach_member_id uuid references public.community_members(id) on delete cascade not null,
  skill_level text not null check (skill_level in ('beginner', 'intermediate', 'advanced')),
  assessed_at timestamptz default now(),
  unique (community_id, subject_member_id, coach_member_id)
);

alter table public.coach_assessments enable row level security;

create policy "coach_assessments_select"
on public.coach_assessments for select to authenticated
using (community_id = ((select auth.jwt()) ->> 'community_id')::uuid);

create policy "coach_assessments_insert"
on public.coach_assessments for insert to authenticated
with check (
  community_id = ((select auth.jwt()) ->> 'community_id')::uuid
  and ((select auth.jwt()) ->> 'user_role') in ('admin', 'coach')
);

create policy "coach_assessments_update"
on public.coach_assessments for update to authenticated

  community_id = ((select auth.jwt()) ->> 'community_id')::uuid
  and ((select auth.jwt()) ->> 'user_role') in ('admin', 'coach')
);

-- 4. progress_notes table
create table public.progress_notes (
  id uuid primary key default gen_random_uuid(),
  community_id uuid references public.communities(id) on delete cascade not null,
  session_id uuid references public.sessions(id) on delete cascade not null,
  subject_member_id uuid references public.community_members(id) on delete cascade not null,
  coach_member_id uuid references public.community_members(id) on delete cascade not null,
  note_text text not null,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique (session_id, subject_member_id, coach_member_id)
);

alter table public.progress_notes enable row level security;

create policy "progress_notes_select"
on public.progress_notes for select to authenticated
using (
  community_id = ((select auth.jwt()) ->> 'community_id')::uuid
  and (
    ((select auth.jwt()) ->> 'user_role') in ('admin', 'coach')
    or subject_member_id = (
      select id from public.community_members

create policy "avatars_update"

**`getProfile(userId?: string)`** -- Fetches profile for given user or current user. Returns `{ success: boolean; data?: PlayerProfile; error?: string }`. Query `player_profiles` filtered by `user_id` and `community_id` from JWT. If no userId passed, use `auth.uid()`. Also fetch the latest `coach_assessments` row for this member.

**`upsertProfile(input: unknown)`** (PROF-01) -- Parse input with `ProfileSchema.parse()`. Auth check: `getUser()` must return a user. Get `community_id` from `getJWTClaims()`. Upsert into `player_profiles` with `onConflict: 'user_id,community_id'`. After successful upsert, sync `display_name` to `community_members` table:
```typescript
await supabase
  .from('community_members')
  .update({ display_name: parsed.displayName })
  .eq('user_id', user.id)
  .eq('community_id', communityId)
```
Call `revalidatePath('/profile')`. Return `{ success: true }` or `{ success: false, error: string }`.

**`setCoachAssessment(input: unknown)`** (PROF-02) -- Parse with `CoachAssessmentSchema.parse()`. Auth check. Role check: `claims.user_role` must be `'admin'` or `'coach'`. Get current user's `community_members.id` as `coach_member_id`. Upsert into `coach_assessments` with `onConflict: 'community_id,subject_member_id,coach_member_id'`. Revalidate `/profile`. Return `ProfileActionResult`.

**`addProgressNote(input: unknown)`** (PROF-04) -- Parse with `ProgressNoteSchema.parse()`. Auth check. Role check: must be `'admin'` or `'coach'`. Get `coach_member_id` from `community_members`. Upsert into `progress_notes` with `onConflict: 'session_id,subject_member_id,coach_member_id'`. Revalidate paths `/profile` and `/coach/sessions/${parsed.sessionId}`. Return `ProfileActionResult`.

**`getLessonHistory(memberId: string, limit = 20, offset = 0)`** (PROF-03) -- Auth check. Query `session_rsvps` joined with `sessions`, `session_coaches` -> `community_members`, and `progress_notes`. Filter: `.eq('member_id', memberId)`, `.eq('rsvp_type', 'confirmed')`, `.is('cancelled_at', null)`. Order by `sessions.scheduled_at` descending. Apply `.range(offset, offset + limit - 1)`. Map results to `LessonHistoryEntry[]`. Return `{ success: boolean; data?: { entries: LessonHistoryEntry[]; summary: LessonHistorySummary }; error?: string }`.

For `LessonHistorySummary`, compute:
- `total_sessions`: count of all confirmed non-cancelled RSVPs for this member (separate count query, no limit)
- `unique_coaches`: count distinct coach member_ids across all sessions attended
- `member_since`: the member's `joined_at` from `community_members`

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> Server Action | Untrusted form input crosses into server-side Supabase operations |
| Browser -> Supabase Storage | File uploads go directly to Supabase from browser client |
| JWT Claims -> RLS | JWT community_id and user_role control all data access |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-01 | Elevation of Privilege | coach_assessments table | mitigate | RLS policy checks `user_role in ('admin', 'coach')` on INSERT/UPDATE; separate table from player_profiles prevents client self-assessment as coach |
| T-3-02 | Information Disclosure | player_profiles table | mitigate | RLS SELECT policy filters `community_id = jwt.community_id` preventing cross-tenant reads |
