Or via the Supabase client:
```bash

const { createClient } = require('@supabase/supabase-js');
const c = createClient(process.env.NEXT_PUBLIC_SUPABASE_URL, process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY);
Promise.all([
  c.from('player_profiles').select('id').limit(0),
  c.from('coach_assessments').select('id').limit(0),
  c.from('progress_notes').select('id').limit(0),
]).then(results => {
  results.forEach((r, i) => console.log(['player_profiles','coach_assessments','progress_notes'][i], r.error ? 'MISSING: ' + r.error.message : 'OK'));
}).catch(console.error);
"
```
