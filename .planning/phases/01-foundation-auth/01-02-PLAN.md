**Step 3 — Replace fonts in `src/app/layout.tsx`:**

Remove `Geist` and `Geist_Mono` imports from `next/font/google`. Replace with:
```typescript
import { Nunito, Nunito_Sans } from 'next/font/google'

const nunito = Nunito({
  subsets: ['latin'],
  weight: ['700'],
  variable: '--font-nunito',
})

const nunitoSans = Nunito_Sans({
  subsets: ['latin'],
  weight: ['400'],
  variable: '--font-nunito-sans',
})
```

Update the `<html>` tag className to use the new font variables:
```tsx
<html lang="en" className={`${nunito.variable} ${nunitoSans.variable}`}>
```

Update `<body>` className to apply the body font:
```tsx
<body className="font-sans antialiased">
```

Create `src/lib/types/auth.ts` with the following type definitions that mirror the database schema and JWT claims. These types are consumed by auth forms, proxy routing, and invite link logic in Plans 03 and 04.

```typescript
// User roles per D-09: Admin, Coach, Client
// 'pending' is the state before community membership is assigned
export type UserRole = 'admin' | 'coach' | 'client' | 'pending'

// Matches community_members table schema
export interface CommunityMember {
  id: string
  community_id: string
  user_id: string
  role: Exclude<UserRole, 'pending'>
  coach_id: string | null
  joined_at: string
}

// Matches communities table schema
export interface Community {
  id: string
  name: string
  slug: string
  created_at: string
}

// Matches invite_links table schema
export interface InviteLink {
  id: string
  community_id: string
  created_by: string
  role: 'coach' | 'client'
  token: string
  revoked_at: string | null
  created_at: string
}

// JWT custom claims injected by Custom Access Token Hook
export interface JWTCustomClaims {
  user_role: UserRole
  community_id: string | null
}

// Auth form state for useActionState pattern
export interface AuthFormState {
  errors?: {
    email?: string[]
    password?: string[]
    general?: string[]
  }
  message?: string
  success?: boolean
}

// Role-based route configuration per D-10
export const ROLE_HOME_ROUTES: Record<Exclude<UserRole, 'pending'>, string> = {
  admin: '/admin',
  coach: '/coach',
  client: '/sessions',
} as const

// Public routes that don't require authentication
export const PUBLIC_ROUTES = ['/auth', '/'] as const
```
